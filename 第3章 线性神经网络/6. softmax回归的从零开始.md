### 1. 初始化参数
```python
import torch

num_inputs = 784  # 28*28
num_outputs = 10

W = torch.normal(0, 0.01, size = (num_inputs, num_outputs), requires_grad = True)  # 权重矩阵
b = torch.zeros(num_outputs, requires_grad = True)  # 偏置向量(懒得随机就全 0)
```

> `train_iter` / `test_iter` 在上一节“图像分类数据集”里加载

---

### 2. softmax 函数
```python
def softmax(X):
    X_exp = torch.exp(X)  # e^x，数值可能会炸所以后面有稳定版
    partition = X_exp.sum(dim = 1, keepdim = True)  # 每一行求和，keepdim 方便广播
    return X_exp / partition  # 每行归一化成概率

# 更稳一点的写法(减 max)
# X_max = X.max(dim = 1, keepdim = True).values
# X_exp = torch.exp(X - X_max)
```

---

### 3. 模型
```python
def net(X):
    X = X.reshape(-1, num_inputs)  # (batch, 1, 28, 28) -> (batch, 784)
    return softmax(X @ W + b)
```

---

### 4. 交叉熵损失
```python
def cross_entropy(y_hat, y):
    # y 是真实类别索引(0~9)，直接取对应概率
    return -torch.log(y_hat[range(len(y_hat)), y])  # 这一行等价于逐行取 y 的那一列
```

---

### 5. 准确率
```python
def accuracy(y_hat, y):
    pred = y_hat.argmax(dim = 1)  # 概率最大的类别
    return (pred == y).float().sum()  # 返回“对了多少个”，方便累计

@torch.no_grad()
def evaluate_accuracy(net, data_iter):
    metric = 0.0         # 计数器: 对了多少个
    num_examples = 0     # 计数器: 一共多少个
    for X, y in data_iter:
        metric += accuracy(net(X), y)  # 每个 batch 累加
        num_examples += y.numel()
    return metric / num_examples
```

---

### 6. 训练 (SGD)
```python
def sgd(params, lr, batch_size):
    with torch.no_grad():
        for param in params:
            param -= lr * param.grad / batch_size  # 梯度是累加的，除以 batch_size 更像均值
            param.grad.zero_()  # 不清零会梯度累积

lr = 0.1
num_epochs = 10

for epoch in range(num_epochs):
    for X, y in train_iter:
        y_hat = net(X)
        l = cross_entropy(y_hat, y)
        l.sum().backward()  # l 是向量，先 sum 再反传
        sgd([W, b], lr, X.shape[0])

    train_acc = evaluate_accuracy(net, train_iter)
    test_acc = evaluate_accuracy(net, test_iter)
    print(f"epoch {epoch + 1}: train_acc {train_acc:.3f}, test_acc {test_acc:.3f}")
```
