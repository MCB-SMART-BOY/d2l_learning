### 1. 造数据(合成一套带噪声的线性关系)
```python
import torch

true_w = torch.tensor([2.0, -3.4])
true_b = 4.2

num_examples = 1000
X = torch.normal(0, 1, (num_examples, 2))
noise = torch.normal(0, 0.01, (num_examples, ))

y = X @ true_w + true_b + noise
# 统一成 (n, 1)，不然下面计算会各种 shape 不对
y = y.reshape(-1, 1)
```

---

### 2. data_iter: 小批量读取
```python
import random

# 属于是固定格式了像 features labels 说实话让我就写 X y (斜眼笑)
def data_iter(batch_size, features, labels):
    num_examples = len(features)
    indices = list(range(num_examples))
    random.shuffle(indices)
    for i in range(0, num_examples, batch_size):
        batch_idx = indices[i: i + batch_size]
        yield features[batch_idx], labels[batch_idx]
```

---

### 3. 定义模型 / 损失 / 优化
```python
# 1) 初始化参数
w = torch.normal(0, 0.01, size = (2, 1), requires_grad = True)
b = torch.zeros(1, requires_grad = True)

# 2) 线性回归模型
def linreg(X, w, b):
    return X @ w + b

# 3) 平方损失
def squared_loss(y_hat, y):
    return (y_hat - y.reshape(y_hat.shape)) ** 2 / 2

# 4) 随机梯度下降 (SGD)
def sgd(params, lr, batch_size):
    with torch.no_grad():
        for param in params:
            param -= lr * param.grad / batch_size
            param.grad.zero_()  # 不清零会累积(第2章的大坑)
```

---

### 4. 训练
```python
lr = 0.03  # 我一开始写的是 0.1 后面觉得太大了 感觉这个比较吃经验
num_epochs = 3
batch_size = 10

for epoch in range(num_epochs):
    for X_batch, y_batch in data_iter(batch_size, X, y):
        l = squared_loss(linreg(X_batch, w, b), y_batch)
        l.sum().backward()  # 不先 sum/mean 的话是对向量求导
        sgd([w, b], lr, batch_size)

    with torch.no_grad():
        train_l = squared_loss(linreg(X, w, b), y).mean()
        print(f"epoch {epoch + 1}, loss {train_l:.6f}")

print("w error:", true_w - w.reshape(true_w.shape))
print("b error:", true_b - b)
```
